    def closeEvent(self, event) -> None:
        """Ablak bezárása — futó mérés esetén safe stop, aztán ablak bezárás."""
        if self._worker is not None and self._worker.is_running():
            reply = QMessageBox.question(
                self,
                "Mérés fut",
                "Mérés fut! Biztonságos leállítás után zárja be az ablakot?\n"
                "(Az összes kimenet le lesz kapcsolva.)",
                QMessageBox.Yes | QMessageBox.No,
                QMessageBox.Yes,
            )
            if reply == QMessageBox.Yes:
                # 1. Emergency stop kérés — a worker belülről hívja safe_all_off()-ot
                self._worker.request_emergency_stop("WINDOW_CLOSED")
                # 2. Várakozás — a worker safe_all_off() után terminál
                if not self._worker.wait(5000):
                    # 3. Fallback: közvetlen hardver-lekapcsolás, ha a worker befagyott
                    #    A hardver biztonságos lekapcsolása fontosabb, mint az azonnali
                    #    ablakbezárás — ezért safe_all_off() a terminate() ELŐTT fut.
                    try:
                        self._instruments.safe_all_off()
                    except Exception:
                        pass
                    # 4. Csak legvégső esetben erőszakos lezárás
                    self._worker.terminate()
                event.accept()
            else:
                event.ignore()
                return
        else:
            event.accept()

         """Impulzus indítása — Rb mérések tick-alapú várakozással."""

  5. Új segédmetódus _run_impulse_wait():
     def _run_impulse_wait(self, dt_s: float, target_s: float,
                           next_state: OcvSocState, measure_slot: str) -> None:
         """Tick-alapú várakozás, majd DMM mérés az adott időpontban."""
         # Terminate voltage ellenőrzés impulzus alatt
         if not self._dmm_valid:
             self._load.input_off()
             self.emergency_stop("DMM_FEEDBACK_LOST_DURING_IMPULSE")
             return
         if self._u_batt < self._profile.terminate_voltage_pack_V * 0.95:
             self._load.input_off()
             self.emergency_stop("TERMINATE_VOLTAGE_DURING_IMPULSE")
             return

         self._impulse_elapsed_s += dt_s
         if self._impulse_elapsed_s < target_s:
             return  # még nem telt el az idő

         # Idő eltelt — mérés
         try:
             v_meas = self._dmm_v.read_voltage()
         except Exception:
             v_meas = self._u_batt  # fallback
         i_set = self._impulse_current_A
         if i_set > 0:
             rb = (self._v_ocv_before_impulse - v_meas) / i_set
         else:
             rb = 0.0

         if measure_slot == "1s":
             self._rb_1s = rb
         elif measure_slot == "10s":
             self._rb_10s = rb
         elif measure_slot == "30s":
             self._rb_30s = rb
             self._load.input_off()

         self._state = next_state
